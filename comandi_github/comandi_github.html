CREDENZIALI

CREATE USER 'github123'@'%' IDENTIFIED BY 'GITHUB';
GRANT SELECT, INSERT, UPDATE, DELETE ON github.* TO br'github'@'%';
FLUSH PRIVILEGES;


PER ACCEDERE AL MIO DB INSERIRE SU http://localhost/phpmyadmin/, il seguente codice sulla console, che si trova in basso:

GRANT ALL PRIVILEGES ON github.* TO 'github123'@'localhost' IDENTIFIED BY 'GITHUB';
FLUSH PRIVILEGES;

$servername = "localhost";
$username = "github123";
$password = "GITHUB";
$dbname = "github";


CREATE TABLE comandi_di_github (
ID_comando INT NOT NULL
AUTO_INCREMENT ,
nome_comando VARCHAR(500) NOT NULL ,
descrizione_comando VARCHAR(500) NOT NULL ,
PRIMARY KEY (ID_comando)
);
